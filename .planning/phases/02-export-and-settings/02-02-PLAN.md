```typescript
function estimateFileSizeKb(width: number, height: number, numFrames: number): number {
  const rawBytes = width * height * numFrames;
  const paletteBytes = numFrames * 768;
  const compressedBytes = rawBytes * 0.5 + paletteBytes + 800;
  return Math.round(compressedBytes / 1024);
}
```
